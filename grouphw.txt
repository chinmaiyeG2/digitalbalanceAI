        // ===== VIEW TOGGLE =====
        function switchView(view) {
            const parent = document.getElementById('parent-view');
            const kid = document.getElementById('kid-view');
            const btnP = document.getElementById('btn-parent');
            const btnK = document.getElementById('btn-kid');
            const stars = document.getElementById('stars-layer');

            if (view === 'parent') {
                parent.classList.remove('hidden');
                kid.classList.add('hidden');
                btnP.classList.add('active');
                btnP.classList.remove('kid-active');
                btnK.classList.remove('active', 'kid-active');
                stars.style.opacity = '0';
                document.body.style.background = '#080c14';
            } else {
                parent.classList.add('hidden');
                kid.classList.remove('hidden');
                btnK.classList.add('active', 'kid-active');
                btnK.classList.remove('active'); // reset then re-add to force style
                btnK.classList.add('kid-active');
                btnP.classList.remove('active');
                stars.style.opacity = '1';
                document.body.style.background = 'linear-gradient(160deg, #0d1117 0%, #1a0d2e 50%, #0d1a1a 100%)';
            }

            // Re-init charts on switch
            setTimeout(() => {
                if (view === 'parent') { initParentCharts(); }
                else { initKidCharts(); }
            }, 50);
        }

        (function generateStars() {
            const layer = document.getElementById('stars-layer');
            for (let i = 0; i < 60; i++) {
                const s = document.createElement('div');
                const size = Math.random() * 2.5 + 0.5;
                s.className = 'star';
                s.style.cssText = `
                    width:${size}px;height:${size}px;
                    left:${Math.random()*100}%;top:${Math.random()*100}%;
                    --dur:${2+Math.random()*4}s;--delay:${Math.random()*3}s;
                    opacity:0;
                `;
                layer.appendChild(s);
            }
        })();
